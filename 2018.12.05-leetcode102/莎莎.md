class Solution:
    def levelOrder(self, root):
        """
        :type root: TreeNode
        :rtype: List[List[int]]
        """
        q = [root]
        v = []
        if not root:
            return []
        while q:
            templist = []
            templen = len(q)
            for i in range(templen):
                temp = q.pop(0)
                templist.append(temp.val)
                if temp.left:
                    q.append(temp.left)
                if temp.right:
                    q.append(temp.right)
            v.append(templist)
        return v
